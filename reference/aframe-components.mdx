## Scene Configuration

### a-scene

The root A-Frame scene entity with atmospheric fog effects.

<ParamField path="fog" type="object">
  Creates the eerie casino atmosphere
  
  <ResponseField name="type" type="string" default="exponential">
    Fog distribution type - uses exponential for dramatic effect
  </ResponseField>
  
  <ResponseField name="color" type="string" default="#0a0000">
    Dark red fog color (#0a0000)
  </ResponseField>
  
  <ResponseField name="density" type="number" default="0.15">
    Fog density value (0.15)
  </ResponseField>
</ParamField>

```html
<a-scene fog="type: exponential; color: #0a0000; density: 0.15">
  <!-- Scene content -->
</a-scene>
```

## Camera & Controls

### Camera Rig

The camera rig positions the player's viewpoint in the scene.

<ParamField path="rig" type="entity">
  Container entity for camera positioning
  
  <ResponseField name="position" type="vec3" default="0 1.6 1.2">
    Positions camera at eye level (1.6m height) slightly forward from table
  </ResponseField>
</ParamField>

### Camera Entity

<ParamField path="camera" type="entity">
  Main camera with look and movement controls
  
  <ResponseField name="camera" type="component">
    Enables first-person perspective
  </ResponseField>
  
  <ResponseField name="look-controls" type="component">
    Mouse/VR headset rotation controls
  </ResponseField>
  
  <ResponseField name="wasd-controls" type="component">
    Keyboard movement controls
    - **acceleration**: 10 (fast movement speed)
  </ResponseField>
  
  <ResponseField name="rotation" type="vec3" default="-25 0 0">
    Tilts camera down 25° to look at table
  </ResponseField>
</ParamField>

```html
<a-entity id="rig" position="0 1.6 1.2">
  <a-entity id="camera" camera look-controls wasd-controls="acceleration: 10" rotation="-25 0 0">
    <!-- Camera children -->
  </a-entity>
</a-entity>
```

### Cursor

<ParamField path="cursor" type="a-cursor">

  <ResponseField name="raycaster" type="component">

  </ResponseField>
  
  <ResponseField name="scale" type="vec3" default="0.5 0.5 0.5">

  </ResponseField>
</ParamField>

### Flickering Point Light

<ParamField path="flicker" type="a-light">
  Animated red light creating horror atmosphere
  
  <ResponseField name="type" type="string" default="point">
    Point light source
  </ResponseField>
  
  <ResponseField name="intensity" type="number" default="1">
    Base intensity
  </ResponseField>
  
  <ResponseField name="position" type="vec3" default="0 3 0">
    Above the table
  </ResponseField>
  
  <ResponseField name="color" type="string" default="#f00">
    Red light (#f00)
  </ResponseField>
  
  <ResponseField name="animation" type="component">
    Flicker animation properties:
    - **property**: intensity
    - **from**: 0.2
    - **to**: 1.2
    - **dur**: 150ms (fast flicker)
    - **loop**: true
    - **dir**: alternate (back and forth)
  </ResponseField>
</ParamField>

```html
<a-light id="flicker" type="point" intensity="1" position="0 3 0" color="#f00"
         animation="property: intensity; from: 0.2; to: 1.2; dur: 150; loop: true; dir: alternate"></a-light>
```

### Card (Dynamic)

Cards are created dynamically in JavaScript:

<ParamField path="card" type="a-box">
  Individual memory card box
  
  <ResponseField name="position" type="vec3">
    Calculated per card:
    - **x**: `(index % 4) * 0.4` (4 columns, 0.4m spacing)
    - **y**: `Math.floor(index / 4) * -0.6` (2 rows, 0.6m spacing)
    - **z**: 0
  </ResponseField>
  
  <ResponseField name="width" type="number" default="0.35">
    Card width (0.35m)
  </ResponseField>
  
  <ResponseField name="height" type="number" default="0.5">
    Card height (0.5m)
  </ResponseField>
  
  <ResponseField name="depth" type="number" default="0.01">
    Very thin card (0.01m)
  </ResponseField>
  
  <ResponseField name="src" type="string" default="#back">
    Initially shows back texture
  </ResponseField>
  
  <ResponseField name="class" type="string" default="clickable">
    Makes card targetable by raycaster
  </ResponseField>
  
  <ResponseField name="data-id" type="string">
    Image identifier (img1, img2, img3, or img4)
  </ResponseField>
  
  <ResponseField name="data-flipped" type="string" default="false">
    Tracks flip state ("true" or "false")
  </ResponseField>
</ParamField>

```javascript
const card = document.createElement('a-box');
card.setAttribute('position', `${x} ${y} 0`);
card.setAttribute('width', '0.35');
card.setAttribute('height', '0.5');
card.setAttribute('depth', '0.01');
card.setAttribute('src', '#back');
card.setAttribute('class', 'clickable');
card.dataset.id = imgId;
card.dataset.flipped = "false";
```

## Audio Components

### Background Music Entity

<ParamField path="bgMusic" type="a-entity">
  Plays looping background music
  
  <ResponseField name="sound.src" type="string" default="#background-music">
    References background audio asset
  </ResponseField>
  
  <ResponseField name="sound.loop" type="boolean" default="true">
    Loops continuously
  </ResponseField>
  
  <ResponseField name="sound.volume" type="number" default="0.3">
    30% volume (not too loud)
  </ResponseField>
</ParamField>

```html
<a-entity id="bgMusic" sound="src: #background-music; loop: true; volume: 0.3"></a-entity>
```

Played via JavaScript:
```javascript
document.querySelector('#bgMusic').components.sound.playSound();
```

### Click Sound

Played dynamically when hovering cards:

```javascript
new Audio(document.querySelector('#click-sound').src).play();
```
